create policy "Users can check their admin role" on public.production_admins for select to authenticated using (email = (select auth.jwt() ->> 'email')); drop policy if exists "Anyone can read production requests" on public.production_ot_requests; drop policy if exists "Anyone can add production requests" on public.production_ot_requests; drop policy if exists "Anyone can update production requests" on public.production_ot_requests; drop policy if exists "Anyone can delete production requests" on public.production_ot_requests; drop policy if exists "Anyone can read production teams" on public.production_teams; drop policy if exists "Anyone can add production team members" on public.production_teams; drop policy if exists "Anyone can delete production team members" on public.production_teams; create policy "Signed in users can read requests" on public.production_ot_requests for select to authenticated using (true); create policy "Signed in users can submit requests" on public.production_ot_requests for insert to authenticated with check (true); create policy "Admins can update requests" on public.production_ot_requests for update to authenticated using (exists (select 1 from public.production_admins where email = (select auth.jwt() ->> 'email'))) with check (exists (select 1 from public.production_admins where email = (select auth.jwt() ->> 'email'))); create policy "Admins can delete requests" on public.production_ot_requests for delete to authenticated using (exists (select 1 from public.production_admins where email = (select auth.jwt() ->> 'email'))); create policy "Signed in users can read teams" on public.production_teams for select to authenticated using (true); create policy "Admins can add team members" on public.production_teams for insert to authenticated with check (exists (select 1 from public.production_admins where email = (select auth.jwt() ->> 'email'))); create policy "Admins can delete team members" on public.production_teams for delete to authenticated using (exists (select 1 from public.production_admins where email = (select auth.jwt() ->> 'email')));
